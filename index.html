Fiche Récapitulative PostgreSQL

Introduction

PostgreSQL est un système de gestion de base de données relationnelle puissant et open source. Voici un récapitulatif des commandes et requêtes SQL les plus courantes.

Commandes de Base

Commande Description Exemple
CREATE DATABASE Créer une nouvelle base de données CREATE DATABASE nom_de_la_base;
CREATE TABLE Créer une nouvelle table CREATE TABLE nom_de_la_table (id SERIAL PRIMARY KEY, nom VARCHAR(100));
INSERT INTO Insérer des données dans une table INSERT INTO nom_de_la_table (nom) VALUES ('Valeur');
SELECT Récupérer des données depuis une table SELECT * FROM nom_de_la_table;
UPDATE Mettre à jour des données dans une table UPDATE nom_de_la_table SET nom = 'Nouvelle Valeur' WHERE id = 1;
DELETE FROM Supprimer des données d'une table DELETE FROM nom_de_la_table WHERE id = 1;

Commandes Avancées

Commande Description Exemple
JOIN Joindre des tables SELECT * FROM table1 JOIN table2 ON table1.id = table2.id;
GROUP BY Grouper des données SELECT COUNT(*), nom FROM table GROUP BY nom;
ORDER BY Ordonner des données SELECT * FROM table ORDER BY nom ASC;
CREATE INDEX Créer un index pour accélérer les recherches CREATE INDEX index_nom ON table(nom);
EXPLAIN Afficher le plan d'exécution d'une requête EXPLAIN SELECT * FROM table;

Types de Données

Type Description Exemple
INTEGER Entier age INTEGER
VARCHAR(n) Chaîne de caractères de longueur variable nom VARCHAR(100)
TEXT Texte de longueur illimitée description TEXT
DATE Date date_naissance DATE
TIMESTAMP Date et heure created_at TIMESTAMP

Transactions

Commande Description Exemple
BEGIN Commencer une transaction BEGIN;
COMMIT Valider une transaction COMMIT;
ROLLBACK Annuler une transaction ROLLBACK;

Vues

Commande Description Exemple
CREATE VIEW Créer une vue CREATE VIEW vue_nom AS SELECT nom, age FROM table;
DROP VIEW Supprimer une vue DROP VIEW vue_nom;

Index

Commande Description Exemple
CREATE INDEX Créer un index CREATE INDEX index_nom ON table(nom);
DROP INDEX Supprimer un index DROP INDEX index_nom;

Fonctions Utiles

Fonction Description Exemple
NOW() Obtenir la date et l'heure actuelles SELECT NOW();
COUNT() Compter le nombre de lignes SELECT COUNT(*) FROM table;
AVG() Calculer la moyenne SELECT AVG(colonne) FROM table;
SUM() Calculer la somme SELECT SUM(colonne) FROM table;
MAX() Obtenir la valeur maximale SELECT MAX(colonne) FROM table;
MIN() Obtenir la valeur minimale SELECT MIN(colonne) FROM table;

Requêtes Complexes

Requête Description Exemple
Subquery Utiliser une sous-requête SELECT nom FROM table WHERE id IN (SELECT id FROM autre_table);
CTE Common Table Expression WITH subquery AS (SELECT * FROM table) SELECT * FROM subquery;
Window Functions Utiliser des fonctions de fenêtre SELECT nom, AVG(salaire) OVER (PARTITION BY departement) FROM employes;
